004package com.ge.predix.mobile.sdk.android;
005
006import android.content.Context;
007import android.content.pm.PackageInfo;
008import android.content.pm.PackageManager;
009import android.os.Build;
010
011import com.ge.predix.mobile.core.VersionAddendumProvider;
012import com.ge.predix.mobile.core.VersionInfo;
013import com.ge.predix.mobile.logging.PredixSDKLogger;
014import com.ge.predix.mobile.platform.PredixConstants;
015
016import java.util.HashMap;
017import java.util.Map;
018
019public final class Version {
020    // the below tokens are replaced during the CI build process. Only the first instance of the token is replaced,
021    // so if needed, additional checking or token comparison can occur further down in the code.
022
023    private static final String buildnumber = "234"; // updated during the CI build process
024    private static final String version = "1.1.234"; // updated during the CI build process
025    private static final String commit = "a904a84fc3272c6372444c5c5cc27556cb7493af"; // updated during the CI build process
026
027    private static String appBuildNumber = "unknown";
028    private static String appVersionNumber = "unknown";
029    private static String appPackageName = "unknown";
030    public static final VersionAddendumProvider PROVIDER = new VersionAddendumProvider() {
031        @Override
032        public Map<String, String> getVersionAddendum() {
033            String buildVersion = com.ge.predix.mobile.sdk.android.Version.appVersionNumber.concat(".").concat(com.ge.predix.mobile.sdk.android.Version.appBuildNumber);
034            Map<String, String> returnMap = new HashMap<String, String>();
035            returnMap.put(PredixConstants.VersionInfoKeys.ApplicationBuildVersion, buildVersion);
036            returnMap.put(PredixConstants.VersionInfoKeys.ApplicationVersion, com.ge.predix.mobile.sdk.android.Version.appVersionNumber);
037            returnMap.put(PredixConstants.VersionInfoKeys.ApplicationBundleId, com.ge.predix.mobile.sdk.android.Version.appPackageName);
038            returnMap.put(PredixConstants.VersionInfoKeys.AndroidSDKVersion, com.ge.predix.mobile.sdk.android.Version.Version);
039            returnMap.put(PredixConstants.VersionInfoKeys.AndroidSDKBuildVersion, com.ge.predix.mobile.sdk.android.Version.BuildVersion);
040            returnMap.put(PredixConstants.VersionInfoKeys.DeviceModel, Build.MODEL);
041            return returnMap;
042        }
043    };
044
045    public static final String Version;
046    public static final String BuildVersion;
047    public static final String Commit;
048
049    private Version() {}
050
051    static {
052        Version = version;
053        BuildVersion = version + "." + buildnumber;
054        Commit = commit;
055
056        addVersionAddendum();
057        new Version();
058    }
059
060    private static void addVersionAddendum() {
061        PredixSDKLogger.trace(Version.getClass().getName(), "Initialized AndroidSDK: Version: " + BuildVersion);
062        VersionInfo.addVersionAddendumProvider(PROVIDER);
063    }
064
065    public static void setupVersionAddendum(Context context) {
066        if (context != null) {
067            PackageManager packageManager = context.getPackageManager();
068            if(packageManager != null) {
069                appPackageName = context.getPackageName();
070
071                try {
072                    PackageInfo packageInfo = packageManager.getPackageInfo(appPackageName, 0);
073                    if (packageInfo != null) {
074                        appVersionNumber = packageInfo.versionName;
075                        appBuildNumber = String.valueOf(packageInfo.versionCode);
076                    }
077                } catch (PackageManager.NameNotFoundException e) {
078                    PredixSDKLogger.error("No package information while gathering version.");
079                }
080            }
081        }
082    }
083}
