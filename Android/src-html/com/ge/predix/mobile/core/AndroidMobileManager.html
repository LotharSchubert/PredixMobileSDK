001package com.ge.predix.mobile.core;
002
003
004import android.content.Context;
005import android.content.res.Configuration;
006import android.support.annotation.NonNull;
007
008import com.ge.predix.mobile.enums.AppLifeCycleEnum;
009import com.ge.predix.mobile.exceptions.InitializationException;
010import com.ge.predix.mobile.platform.CookieStorage;
011import com.ge.predix.mobile.sdk.android.logging.LogCatLogger;
012import com.ge.predix.mobile.serviceFoundation.ServiceRouter;
013import com.ge.predix.mobile.services.OrientationChangeManagerImpl;
014
015import dagger.ObjectGraph;
016
017/**
018 * Created by 212460388.
019 * Copyright Â© 2016 GE. All rights reserved.
020 * <p>
021 * This class is the Android representation of the {@link MobileManager} found in the JavaSDK.
022 * This class should be used on Android instead of the {@link MobileManager}. It contains analogous methods
023 * to the {@link MobileManager} and uses it internally.
024 * The {@link MobileManager} is the core of the Predix Mobile SDK system. It is the component used to bind your platform
025 * user interface to the internals of the SDK. Your container is responsible for providing a window or view where authentication, is loaded,
026 * and a separate window or view to load the PM web app into. The manager can be used to register custom services prior to startup.
027 * It can also be used to obtain the core component responsible for distributing notifications.
028 * </p>
029 */
030public class AndroidMobileManager implements AppLifeCycle {
031
032    public static AndroidMobileManager instance = new AndroidMobileManager();
033    private AppStateManager stateManager;
034    private MobileManager mobileManager;
035    private AndroidMemChecker androidMemChecker;
036
037    @NonNull
038    private Injector getAndroidInjector(final Context context) {
039        return new Injector() {
040
041            @Override
042            public void inject(Object mainObject, ViewInterface dependencies) {
043                final AndroidMobileManagerModule androidMobileManagerModule = new AndroidMobileManagerModule(context, dependencies);
044                final ObjectGraph objectGraph = ObjectGraph.create(androidMobileManagerModule);
045                androidMobileManagerModule.setGraph(objectGraph);
046                objectGraph.inject(mainObject);
047            }
048        };
049    }
050
051    AndroidMobileManager() {
052        if (null == stateManager) {
053            stateManager = new AppStateManager();
054        }
055    }
056
057    /**
058     * Use this method to obtain an instance of the {@link ServiceRouter} which can be used to register additional
059     * Predix Mobile services. A Predix Mobile Service is a Java object that responds to an http://pmapi RESTful call.
060     * These objects are decorated with {@link com.ge.predix.mobile.annotations.RequestMapping} annotations that
061     * dictate which objects and methodsrespond to which RESTful calls.
062     * @return an instance of the {@link ServiceRouter}
063     */
064    public ServiceRouter serviceRouter() {
065        return instance.mobileManager.serviceRouter;
066    }
067
068    /**
069     * This method starts the mobile manager and invokes the equivalent of a http://pmapi/boot/start REST command.
070     * Configuration is read in from the preferences and config files.
071     * The internal service router is instantiated and all core and custom services will be registered.
072     * @throws InitializationException Thrown if start is called when the manager is in the middle of stopping.
073     */
074    public void start() throws InitializationException {
075        mobileManager.start();
076        this.androidMemChecker.startMemChecker();
077    }
078
079    public void stop() throws InitializationException {
080        mobileManager.stop();
081        mobileManager = null;
082        this.androidMemChecker.stopMemChecker();
083    }
084
085    public boolean isRunning() {
086        return null != this.mobileManager && this.mobileManager.getState() == MobileManager.PMMState.STARTED;
087    }
088
089    CookieStorage cookieStorage() {
090        return mobileManager.cookieStorage;
091    }
092
093    MobileManager mobileManager() {
094        return mobileManager;
095    }
096
097    /**
098     * {@link Deprecated} use the instance field directly followed by a separate call to initialize.
099     * Use this method to obtain an instance of the {@link AndroidMobileManager}
100     * @param context An Android SDK {@link Context} object which is used internally by the SDK.
101     * @param viewInterface A {@link ViewInterface} which is an abstraction around the component used to link Predix Mobile SDK behaviors to
102     * your application. It contains methods that allow you to bind an {@link AuthHandler} component, a
103     * {@link com.ge.predix.mobile.platform.WindowView} component, the platform's context and a system preferences component.
104     * @return an initialized instance of the {@link AndroidMobileManager}
105     */
106    @Deprecated
107    public static AndroidMobileManager instance(Context context, ViewInterface viewInterface) {
108        AndroidMobileManager.instance.initialize(context, viewInterface);
109        return AndroidMobileManager.instance;
110    }
111
112    /**
113     * Initializes the {@link AndroidMobileManager} instance with the given Android SDK {@link Context} and {@link ViewInterface}
114     * @param context An Android SDK {@link Context} object which is used internally by the SDK.
115     * @param viewInterface A {@link ViewInterface} which is an abstraction around the component used to link Predix Mobile SDK behaviors to
116     * your application. It contains methods that allow you to bind an {@link AuthHandler} component, a
117     * {@link com.ge.predix.mobile.platform.WindowView} component, the platform's context and a system preferences component.
118     */
119    public void initialize(Context context, final ViewInterface viewInterface) {
120        if (mobileManager == null) {
121            PredixMobileConfiguration.baseAssetFolderPath = "android_asset";
122            mobileManager = MobileManager.instance();
123            mobileManager.setInjector(getAndroidInjector(context));
124            mobileManager.setViewInterface(viewInterface);
125            mobileManager.setLogger(new LogCatLogger());
126            com.ge.predix.mobile.sdk.android.Version.setupVersionAddendum(context);
127            this.androidMemChecker = new AndroidMemChecker(context);
128        }
129    }
130
131    @Override
132    public void onStart() {
133        this.stateManager.handleStateChange(AppLifeCycleEnum.Start);
134    }
135
136    @Override
137    public void onResume() {
138        this.stateManager.handleStateChange(AppLifeCycleEnum.Resume);
139    }
140
141    @Override
142    public void onPause() {
143        this.stateManager.handleStateChange(AppLifeCycleEnum.Pause);
144    }
145
146    @Override
147    public void onStop() {
148        this.stateManager.handleStateChange(AppLifeCycleEnum.Stop);
149    }
150
151    @Override
152    public void onDestroy() {
153        this.stateManager.handleStateChange(AppLifeCycleEnum.Destroy);
154    }
155
156    @Override
157    public void onConfigurationChanged(Object config) {
158        if (config instanceof Configuration) {
159            Configuration configuration = (Configuration) config;
160            OrientationChangeManagerImpl.getInstance().handleOrientationChange(configuration.orientation);
161        }
162    }
163}
