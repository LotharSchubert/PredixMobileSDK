020/**
021 * Created by 212460388.
022 * Copyright Â© 2016 GE. All rights reserved.
023 * <p>
024 * This class is the Android representation of the {@link MobileManager} found in the JavaSDK.
025 * This class should be used on Android instead of the {@link MobileManager}. It contains analogous methods
026 * to the {@link MobileManager} and uses it internally.
027 * The {@link MobileManager} is the core of the Predix Mobile SDK system. It is the component used to bind your platform
028 * user interface to the internals of the SDK. Your container is responsible for providing a window or view where authentication, is loaded,
029 * and a separate window or view to load the PM web app into. The manager can be used to register custom services prior to startup.
030 * It can also be used to obtain the core component responsible for distributing notifications.
031 * </p>
032 */
033public class AndroidMobileManager implements AppLifeCycle {
034
035    private static volatile AndroidMobileManager instance = null;
036    private AppStateManager stateManager;
037    private MobileManager mobileManager;
038    private DeviceStorageBroadcastReceiver storageBroadcastReceiver;
039    private Context context;
040    private LowMemoryNotificationsReceiverImpl lowMemReceiver;
041
042    @NonNull
043    private DaggerGraphManager getAndroidInjector(final Context context) {
044        return new DaggerGraphManager() {
045            AndroidMobileManagerModule androidMobileManagerModule;
046            ObjectGraph objectGraph;
047
048            @Override
049            void updateAuthenticationManager(AuthenticationManager authenticationManager) {
050                getApplicationGraph();
051                androidMobileManagerModule.setAuthenticationManager(authenticationManager);
052            }
053
054            @Override
055            void updateViewInterface(ViewInterface viewInterface) {
056                getApplicationGraph();
057                androidMobileManagerModule.setViewInterface(viewInterface);
058            }
059
060            @Override
061            protected ObjectGraph getApplicationGraph() {
062                if (objectGraph == null) {
063                    androidMobileManagerModule = new AndroidMobileManagerModule(context);
064                    objectGraph = ObjectGraph.create(androidMobileManagerModule);
065                    androidMobileManagerModule.setGraph(objectGraph);
066                }
067
068                return objectGraph;
069            }
070        };
071    }
072
073    private AndroidMobileManager() {
074        if (null == stateManager) {
075            stateManager = new AppStateManager();
076        }
077    }
078
079    public static AndroidMobileManager getInstance() {
080
081        if (instance == null) {
082            synchronized (AndroidMobileManager.class) {
083                if (instance == null) {
084                    instance = new AndroidMobileManager();
085                }
086            }
087        }
088        return instance;
089    }
090
091    /**
092     * Use this method to obtain an instance of the {@link ServiceRouter} which can be used to register additional
093     * Predix Mobile services. A Predix Mobile Service is a Java object that responds to an http://pmapi RESTful call.
094     * These objects are decorated with {@link com.ge.predix.mobile.annotations.RequestMapping} annotations that
095     * dictate which objects and methodsrespond to which RESTful calls.
096     * @return an instance of the {@link ServiceRouter}
097     */
098    public ServiceRouter serviceRouter() {
099        return instance.mobileManager.serviceRouter;
100    }
101
102    /**
103     * This method starts the mobile manager and invokes the equivalent of a http://pmapi/boot/start REST command.
104     * Configuration is read in from the preferences and config files.
105     * The internal service router is instantiated and all core and custom services will be registered.
106     * @throws InitializationException Thrown if start is called when the manager is in the middle of stopping.
107     */
108    public void start() throws InitializationException {
109        mobileManager.start();
110    }
111
112    public void stop() throws InitializationException {
113        mobileManager.stop();
114        mobileManager.cleanup();
115        mobileManager = null;
116        cleanup();
117    }
118
119    private void cleanup() {
120        instance = null;
121    }
122
123    public boolean isRunning() {
124        return null != this.mobileManager && this.mobileManager.getState() == MobileManager.PMMState.STARTED;
125    }
126
127    CookieStorage cookieStorage() {
128        return mobileManager.cookieStorage;
129    }
130
131    MobileManager mobileManager() {
132        return mobileManager;
133    }
134
135    /**
136     * {@link Deprecated} use the instance field directly followed by a separate call to initialize.
137     * Use this method to obtain an instance of the {@link AndroidMobileManager}
138     * @param context An Android SDK {@link Context} object which is used internally by the SDK.
139     * @param viewInterface A {@link ViewInterface} which is an abstraction around the component used to link Predix Mobile SDK behaviors to
140     * your application. It contains methods that allow you to bind an {@link AuthHandler} component, a
141     * {@link com.ge.predix.mobile.platform.WindowView} component, the platform's context and a system preferences component.
142     * @return an initialized instance of the {@link AndroidMobileManager}
143     */
144    @Deprecated
145    public static AndroidMobileManager instance(Context context, ViewInterface viewInterface) {
146        AndroidMobileManager.instance.initialize(context, viewInterface);
147        return AndroidMobileManager.instance;
148    }
149
150    /**
151     * Initializes the {@link AndroidMobileManager} instance with the given Android SDK {@link Context} and {@link ViewInterface}
152     * @param context An Android SDK {@link Context} object which is used internally by the SDK.
153     * @param viewInterface A {@link ViewInterface} which is an abstraction around the component used to link Predix Mobile SDK behaviors to
154     * your application. It contains methods that allow you to bind an {@link AuthHandler} component, a
155     * {@link com.ge.predix.mobile.platform.WindowView} component, the platform's context and a system preferences component.
156     */
157    public void initialize(Context context, final ViewInterface viewInterface) {
158        this.context = context;
159        if (mobileManager == null) {
160            PredixMobileConfiguration.baseAssetFolderPath = "android_asset";
161            DaggerGraphManager injector = getAndroidInjector(context);
162            mobileManager = MobileManager.instance();
163            mobileManager.setViewInterface(viewInterface);
164            mobileManager.setDaggerGraphManager(injector);
165            mobileManager.setLogger(new LogCatLogger());
166            com.ge.predix.mobile.sdk.android.Version.setupVersionAddendum(context);
167            this.lowMemReceiver = new LowMemoryNotificationsReceiverImpl();
168            this.storageBroadcastReceiver = new DeviceStorageBroadcastReceiver();
169        }
170    }
171
172    @Override
173    public void onStart() {
174        this.stateManager.handleStateChange(AppLifeCycleEnum.Start);
175    }
176
177    @Override
178    public void onResume() {
179        this.stateManager.handleStateChange(AppLifeCycleEnum.Resume);
180        this.registerStorageBroadcastReceiver();
181    }
182
183    @Override
184    public void onPause() {
185        this.stateManager.handleStateChange(AppLifeCycleEnum.Pause);
186        this.unregisterStorageBroadcastReceiver();
187    }
188
189    @Override
190    public void onStop() {
191        this.stateManager.handleStateChange(AppLifeCycleEnum.Stop);
192    }
193
194    @Override
195    public void onDestroy() {
196        this.stateManager.handleStateChange(AppLifeCycleEnum.Destroy);
197    }
198
199    @Override
200    public void onConfigurationChanged(Object config) {
201        if (config instanceof Configuration) {
202            Configuration configuration = (Configuration) config;
203            OrientationChangeManagerImpl.getInstance().handleOrientationChange(configuration.orientation);
204        }
205    }
206
207    /**
208     * A pass-through of onLowMemory signal to PredixMobile sdk.
209     */
210    public void onLowMemory() {
211        this.lowMemReceiver.onLowMemory();
212    }
213
214    /**
215     * A pass-through of onTrimMemory level signal to PredixMobile sdk.
216     * @param level The context of the trim, giving a hint of the amount of
217     * trimming the application may like to perform.
218     */
219    public void onTrimMemory(int level) {
220        this.lowMemReceiver.onTrimMemory(level);
221    }
222
223    private void registerStorageBroadcastReceiver() {
224        PredixSDKLogger.trace(this, "Registering DeviceStorageBroadcastReceiver.");
225        IntentFilter filter = new IntentFilter();
226        filter.addAction(Intent.ACTION_DEVICE_STORAGE_OK);
227        filter.addAction(Intent.ACTION_DEVICE_STORAGE_LOW);
228
229        this.context.registerReceiver(this.storageBroadcastReceiver, filter);
230    }
231
232    private void unregisterStorageBroadcastReceiver() {
233        PredixSDKLogger.trace(this, "Un-registering DeviceStorageBroadcastReceiver.");
234        this.context.unregisterReceiver(this.storageBroadcastReceiver);
235    }
