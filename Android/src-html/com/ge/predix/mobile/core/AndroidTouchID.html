001package com.ge.predix.mobile.core;
002
003import android.annotation.TargetApi;
004import android.app.Activity;
005import android.content.Context;
006import android.content.DialogInterface;
007import android.hardware.fingerprint.FingerprintManager;
008import android.os.Build;
009import android.support.v7.app.AlertDialog;
010
011import com.ge.predix.mobile.logging.PredixSDKLogger;
012import com.ge.predix.mobile.sdk.R;
013
014import javax.inject.Inject;
015
016/**
017 * Created by George Fouche on 2/16/17.
018 */
019
020public class AndroidTouchID implements TouchID {
021
022    @Inject
023    public Context context;
024    private String supportedReason;
025    private TouchIDAuth touchIDAuth;
026    private TouchIDAuthHandler touchIDAuthHandler;
027    private AlertDialog dialog;
028    private  static boolean isCurrentlyValidating = false;
029
030
031    public AndroidTouchID() {
032        touchIDAuthHandler = new TouchIDAuthHandler();
033    }
034
035    @Override
036    public boolean isSupported() {
037        touchIDAuth = new TouchIDAuth(context);
038        try {
039            if (!touchIDAuth.isFingerprintSupported()) {
040                supportedReason = touchIDAuth.getMessage();
041                return false;
042
043            }
044        } catch (TouchIDAuthException e) {
045            PredixSDKLogger.trace(this, e.getMessage());
046            return false;
047        }
048        supportedReason = touchIDAuth.getMessage();
049        return true;
050    }
051
052
053    @TargetApi(Build.VERSION_CODES.M)
054    @Override
055    public void validate(String promptMessage, final TouchIDCallback touchIDCallback) {
056        if (isCurrentlyValidating) {
057            touchIDCallback.onComplete(true, false, "Validation is in Progress");
058            return;
059        } else {
060            isCurrentlyValidating = true;
061            doValidation(promptMessage, touchIDCallback);
062        }
063
064    }
065
066    @Override
067    public String getSupportedReason() {
068        return supportedReason;
069    }
070
071
072    @TargetApi(Build.VERSION_CODES.M)
073    private void doValidation(String promptMessage, final TouchIDCallback touchIDCallback) {
074        PredixSDKLogger.trace(this, "Start fingerprint validation");
075        touchIDAuth = new TouchIDAuth(context);
076
077        if (isSupported()) {
078            try {
079                touchIDAuth.setupKeyAndCipher();
080            } catch (TouchIDAuthException e) {
081                PredixSDKLogger.error(this, e.getMessage());
082            }
083
084            FingerprintManager fingerprintManager = touchIDAuth.getFingerprintManager();
085            FingerprintManager.CryptoObject cryptoObject = touchIDAuth.getCryptoObject();
086            showDialog(promptMessage);
087            touchIDAuthHandler.doAuth(fingerprintManager, cryptoObject, new TouchIDCallback() {
088                @Override
089                public void onComplete(boolean supported, boolean isOwner, String message) {
090                    hideDialog();
091                    touchIDCallback.onComplete(supported, isOwner, message);
092                    isCurrentlyValidating = false;
093                }
094            });
095        } else {
096            touchIDCallback.onComplete(false, false, supportedReason);
097            isCurrentlyValidating = false;
098        }
099        PredixSDKLogger.trace(this, "Fingerprint validation completed");
100    }
101
102    @TargetApi(Build.VERSION_CODES.M)
103    private void showDialog(final String message) {
104        ((Activity) context).runOnUiThread(new Runnable() {
105            public void run() {
106                AlertDialog.Builder builder = new AlertDialog.Builder(context);
107                if (dialog != null) hideDialog();
108
109                builder.setMessage(message).setTitle("Authenticate").setIcon(R.drawable.ic_fp_40px).setCancelable(true)
110                        .setNegativeButton("Cancel", new DialogInterface.OnClickListener() {
111                            @Override
112                            public void onClick(DialogInterface dialog, int which) {
113                                touchIDAuthHandler.cancelAuth();
114                                dialog.cancel();
115                            }
116                        });
117
118                // creating dialog box
119                dialog = builder.create();
120                dialog.show();
121
122            }
123        });
124    }
125
126
127    private void hideDialog() {
128        dialog.hide();
129        dialog = null;
130    }
131
132}
