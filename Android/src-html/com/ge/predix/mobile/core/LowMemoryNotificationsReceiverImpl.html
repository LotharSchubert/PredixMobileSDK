001package com.ge.predix.mobile.core;
002
003import android.content.ComponentCallbacks2;
004import android.content.res.Configuration;
005
006import com.ge.predix.mobile.core.notifications.LowMemoryNotification;
007import com.ge.predix.mobile.enums.MemoryNotificationLevel;
008import com.ge.predix.mobile.logging.PredixSDKLogger;
009
010/**
011 * Created by 212460388 on 6/23/17.
012 */
013
014public class LowMemoryNotificationsReceiverImpl implements ComponentCallbacks2, LowMemoryNotificationsReceiver {
015    @Override
016    public void onLowMemory(MemoryNotificationLevel memoryNotificationLevel) {
017        sendNotification(memoryNotificationLevel);
018    }
019
020    @Override
021    public void onTrimMemory(int level) {
022        switch (level) {
023
024//               Release any UI objects that currently hold memory. The user interface has moved to the background.
025//               NotificationBus.defaultInstance().postNotification(new LowMemoryNotification(MemoryNotificationLevel.Normal));
026            case ComponentCallbacks2.TRIM_MEMORY_UI_HIDDEN:
027
028            case ComponentCallbacks2.TRIM_MEMORY_RUNNING_MODERATE:
029            case ComponentCallbacks2.TRIM_MEMORY_RUNNING_LOW:
030            case ComponentCallbacks2.TRIM_MEMORY_RUNNING_CRITICAL:
031//               Release any memory that your app doesn't need to run. The device is running low on memory while the app is running.
032//               The event raised indicates the severity of the memory-related event. If the event is TRIM_MEMORY_RUNNING_CRITICAL, then the system will begin killing background processes.
033                onLowMemory(MemoryNotificationLevel.Warning);
034                break;
035
036            case ComponentCallbacks2.TRIM_MEMORY_BACKGROUND:
037            case ComponentCallbacks2.TRIM_MEMORY_MODERATE:
038            case ComponentCallbacks2.TRIM_MEMORY_COMPLETE:
039
040//               Release as much memory as the process can. The app is on the LRU list and the system is running low on memory.
041//               The event raised indicates where the app sits within the LRU list. If the event is TRIM_MEMORY_COMPLETE, the process will be one of the first to be terminated.
042                onLowMemory(MemoryNotificationLevel.Critical);
043                break;
044
045            default:
046//                Release any non-critical data structures. The app received an unrecognized memory level value from the system. Treat this as a generic low-memory message.
047                onLowMemory(MemoryNotificationLevel.Any);
048                break;
049        }
050    }
051
052    @Override
053    public void onConfigurationChanged(Configuration newConfig) {
054
055    }
056
057    @Override
058    public void onLowMemory() {
059        PredixSDKLogger.warning(this, "A critical memory warning!");
060        onLowMemory(MemoryNotificationLevel.Critical);
061    }
062
063    private void sendNotification(MemoryNotificationLevel notificationLevel) {
064        PredixSDKLogger.warning(this, "Sending a low-memory notification: "+notificationLevel.toString());
065        NotificationBus.defaultInstance().postNotification(new LowMemoryNotification(notificationLevel));
066    }
067}
