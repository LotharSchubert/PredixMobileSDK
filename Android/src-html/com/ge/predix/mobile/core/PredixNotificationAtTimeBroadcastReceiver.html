031public abstract class PredixNotificationAtTimeBroadcastReceiver extends BroadcastReceiver {
032
033    /**
034     * Should provide a Notification object which will be shown in notification bar when a notification_at_time event will be fired and application will be in background.
035     * @param context
036     * @param intent
037     * @param NOTIFICATION_PROMPT_KEY
038     * @param NOTIFICATION_ADDITIONAL_KEY
039     * @return
040     */
041    public abstract Notification createNotification(final Context context, final Intent intent, final String NOTIFICATION_PROMPT_KEY, final String NOTIFICATION_ADDITIONAL_KEY);
042
043    private boolean appForeground = false;
044    public final int requestCode = 91916; //09-19-2016
045
046    public void onReceive(Context context, Intent intent) {
047//        this.alarmManager = (AlarmManager) context.getSystemService(Context.ALARM_SERVICE);
048        if (!intent.getAction().equals(DISPLAY_ACTION)){
049            PredixSDKLogger.trace(this, "Received an unknown intent: "+intent.getAction());
050            return;
051        }
052
053        appForeground = isAppForeground(context);
054        if (appForeground) {
055            PredixSDKLogger.trace(this, "App is in foreground. Sending attime notification.");
056            Bundle bundle = intent.getExtras();
057            JSONObject notificationResponse ;
058            notificationResponse = new JSONObject();
059            try {
060                notificationResponse.put(NOTIFICATION_PROMPT_KEY, bundle.getString(NOTIFICATION_PROMPT_KEY));
061                String scriptString = bundle.getString(NOTIFICATION_JS_CB_FUNC_KEY);
062                if(scriptString != null && scriptString.length() > 0) {
063                    notificationResponse.put(NOTIFICATION_JS_CB_FUNC_KEY, scriptString);
064                }
065                notificationResponse.put(NOTIFICATION_DATE_KEY, bundle.getString(NOTIFICATION_DATE_KEY));
066                notificationResponse.put(NOTIFICATION_ADDITIONAL_KEY, bundle.getString(NOTIFICATION_ADDITIONAL_KEY));
067                notificationResponse.put(NOTIFICATION_ID_KEY, bundle.getInt(NOTIFICATION_ID_KEY));
068            } catch (JSONException je) {
069                PredixSDKLogger.error(this, "Error in parsing a attime notification!", je);
070            }
071            AtTimeNotification pmNotification = new AtTimeNotification(notificationResponse);
072            NotificationBus.defaultInstance().postNotification(pmNotification);
073        }
074        else { //wake up buddy
075            PredixSDKLogger.debug(this, "App is in background. Creating a local notification to wake up.");
076            NotificationManager notificationManager = (NotificationManager) context.getSystemService(Context.NOTIFICATION_SERVICE);
077            Notification userNotification = this.createNotification(context, intent, NOTIFICATION_PROMPT_KEY, NOTIFICATION_ADDITIONAL_KEY);
078            if (null != userNotification && notificationManager != null) {
079                notificationManager.notify((int) System.currentTimeMillis(), userNotification);
080            }
081            else {
082                PredixSDKLogger.trace(this, "I was in unit test mode.");
083            }
084        }
085    }
086
087    private boolean isAppForeground(Context context) {
088
089        ActivityManager activityManager = (ActivityManager) context.getSystemService(Context.ACTIVITY_SERVICE);
090        List<ActivityManager.RunningAppProcessInfo> appProcesses = activityManager.getRunningAppProcesses();
091        if (appProcesses == null) {
092            return false;
093        }
094        final String packageName = context.getPackageName();
095        for (ActivityManager.RunningAppProcessInfo appProcess : appProcesses) {
096            if (appProcess.importance == ActivityManager.RunningAppProcessInfo.IMPORTANCE_FOREGROUND && appProcess.processName.equals(packageName)) {
097                return true;
098            }
099        }
100        return false;
101    }
102
103    final private class AtTimeNotification extends NotificationBase {
104        public AtTimeNotification(JSONObject payload) {
105            super(payload);
106        }
107        @Override
108        public String getName() {
109//          subscriber should use this name to listen
110            return ATTIME_NOTIFICATION_KEY;
111        }
112    }
