008package com.ge.predix.mobile.core;
009
010import android.content.Context;
011import android.graphics.Bitmap;
012import android.webkit.ValueCallback;
013import android.webkit.WebResourceRequest;
014import android.webkit.WebResourceResponse;
015import android.webkit.WebView;
016import android.webkit.WebViewClient;
017
018import com.ge.predix.mobile.logging.PredixSDKLogger;
019import com.ge.predix.mobile.network.URLResponse;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.util.Map;
024import java.util.concurrent.LinkedBlockingDeque;
025
026/**
027 * Provides an interface that bridges the Android WebView and the Predix Mobile Android SDK
028 * <p>
029 * It is not required that an application use the WebViewClient, however if an application wishes to implement
030 * there on version of the WebViewClient they will be required to wire the features they want from the SDK
031 * to the WebView.
032 * <p>
033 * Note:  When overriding this class be sure to call super on your method implementations unless you intend
034 * to override the default functionality.
035 */
036
037public class ServiceRouterWebViewClient extends WebViewClient {
038    RequestProcessor requestProcessor = new RequestProcessor(Platform.instance);
039    private static final String ProcessHeader = "pmapi-body";
040    private LinkedBlockingDeque<String> bridgeQueue = new LinkedBlockingDeque<>();
041
042    @Override
043    public void onLoadResource(WebView view, String url) {
044        super.onLoadResource(view, url);
045        this.injectJavaScriptBridgeIfNotLoaded(view, url);
046    }
047
048    @Override
049    public void onPageStarted(WebView view, String url, Bitmap favicon) {
050        super.onPageStarted(view, url, favicon);
051        this.injectJavaScriptBridgeIfNotLoaded(view, url);
052    }
053
054    @Override
055    public void onPageFinished(WebView view, String url) {
056        super.onPageFinished(view, url);
057        this.injectJavaScriptBridgeIfNotLoaded(view, url);
058    }
059
060    @Override
061    public WebResourceResponse shouldInterceptRequest(WebView view, WebResourceRequest request) {
062        String url = request.getUrl().toString();
063        if (this.requestProcessor.canProcessRequest(url)) {
064            String method = request.getMethod();
065            Map<String, String> headers = request.getRequestHeaders();
066            String body = headers.get(ProcessHeader);
067            headers.put(ProcessHeader, ""); //Remove the header data but don't remove the header because of CORS
068            URLResponse response = requestProcessor.processRequest(url, method, body, headers);
069            if (response != null) {
070                return new AndroidWebResource(response);
071            } else {
072                PredixSDKLogger.trace(this, "shouldInterceptRequest response to: url: " + request.getUrl() + " with status: " + -1);
073                return new WebResourceResponse("text/plain", "UTF-8", null);
074            }
075        } else if ("pmapi.js".equals(request.getUrl().getLastPathSegment())) {
076            try {
077                return new WebResourceResponse("text/javascript", "UTF-8", loadInputStream(view.getContext()));
078            } catch (IOException e) {
079                return super.shouldInterceptRequest(view, request);
080            }
081        } else {
082            return super.shouldInterceptRequest(view, request);
083        }
084    }
085
086    @Override
087    public boolean shouldOverrideUrlLoading(WebView view, String url) {
088        if (this.requestProcessor.urlIsAuthenticationCallback(url)) {
089            this.requestProcessor.processRequest(url, "GET", null, null);
090            return true;
091        }
092        return super.shouldOverrideUrlLoading(view, url);
093    }
094
095    private synchronized void injectJavaScriptBridgeIfNotLoaded(final WebView view, final String url) {
096        if (url == null || !url.contains(".html") || bridgeQueue.contains(url)) return;
097        bridgeQueue.add(url);
098        view.evaluateJavascript("document.getElementById('pmapiJSBridge') != null", new ValueCallback<String>() {
099            @Override
100            public void onReceiveValue(String value) {
101                boolean bridgeLoaded = value != null && Boolean.valueOf(value);
102                PredixSDKLogger.trace(this, "Javascript bridge is " + ((bridgeLoaded) ? "" : "not ") + "loaded");
103                if (bridgeLoaded) {
104                    bridgeQueue.remove(url);
105                    return;
106                }
107                PredixSDKLogger.trace(this, "loading javascript bridge");
108                injectBridgeIntoPage(view, url);
109            }
110        });
111
112    }
113
114    private void injectBridgeIntoPage(final WebView view, final String url) {
115        view.evaluateJavascript("javascript:(function() {" +
116                "var parent = document.getElementsByTagName('body').item(0);" +
117                "var script = document.createElement('script');" +
118                "script.setAttribute('src', 'pmapi.js');" +
119                "script.setAttribute('id', 'pmapiJSBridge');" +
120                "parent.appendChild(script)" +
121                "})()", new ValueCallback<String>() {
122            @Override
123            public void onReceiveValue(String value) {
124                bridgeQueue.remove(url);
125            }
126        });
127        PredixSDKLogger.trace(this, "javascript bridge loaded into webview");
128    }
129
130    InputStream loadInputStream(Context context) throws IOException {
131        return context.getAssets().open("www/pmapi.js");
132    }
133}
134
135class AndroidWebResource extends WebResourceResponse {
136    private int statusCode;
137    private String reasonPhrase;
138
139    public AndroidWebResource(URLResponse response) {
140        super(response.contentType(), response.encoding(), response.statusCode(), response.statusPhrase(), response.getFlatHeaders(), response.bodyInputStream());
141    }
142
143    @Override
144    public void setStatusCodeAndReasonPhrase(int statusCode, String reasonPhrase) {
145        this.statusCode = statusCode;
146        this.reasonPhrase = reasonPhrase;
147    }
148
149    @Override
150    public int getStatusCode() {
151        return this.statusCode;
152    }
153
154    @Override
155    public String getReasonPhrase() {
156        return this.reasonPhrase;
157    }
158}
