001package com.ge.predix.mobile.core;
002
003import android.annotation.TargetApi;
004import android.app.KeyguardManager;
005import android.content.Context;
006import android.hardware.fingerprint.FingerprintManager;
007import android.os.Build;
008import android.security.keystore.KeyGenParameterSpec;
009import android.security.keystore.KeyProperties;
010
011import com.ge.predix.mobile.logging.PredixSDKLogger;
012
013import java.io.IOException;
014import java.security.InvalidAlgorithmParameterException;
015import java.security.InvalidKeyException;
016import java.security.KeyStore;
017import java.security.KeyStoreException;
018import java.security.NoSuchAlgorithmException;
019import java.security.NoSuchProviderException;
020import java.security.UnrecoverableKeyException;
021import java.security.cert.CertificateException;
022
023import javax.crypto.Cipher;
024import javax.crypto.KeyGenerator;
025import javax.crypto.NoSuchPaddingException;
026import javax.crypto.SecretKey;
027
028import static android.content.Context.FINGERPRINT_SERVICE;
029import static android.content.Context.KEYGUARD_SERVICE;
030

034
035public class TouchIDAuth {
036    private String message;
037    private static final String KEY_NAME = "fingerprintKey";
038    private static final String ANDROID_KEYSTORE_NAME = "AndroidKeyStore";
039
040
041    private KeyStore keyStore;
042    private KeyGenerator keyGenerator;
043    private Context context;
044    private FingerprintManager fingerprintManager;
045    private FingerprintManager.CryptoObject cryptoObject;
046
047    public TouchIDAuth(Context context) {
048        this.context = context;
049    }
050
051    @TargetApi(Build.VERSION_CODES.M)
052    public boolean isFingerprintSupported() throws TouchIDAuthException {
053
054
055        KeyguardManager keyguardManager = (KeyguardManager) context.getSystemService(KEYGUARD_SERVICE);
056        fingerprintManager = (FingerprintManager) context.getSystemService(FINGERPRINT_SERVICE);
057
058        try {
059            // Check if the fingerprint sensor is present
060            if (!fingerprintManager.isHardwareDetected()) {
061                message = "Fingerprint authentication not supported";
062                PredixSDKLogger.trace(this, message);
063                return false;
064            }
065            // Verify that at least one fingerprint is registered on the device
066            if (!fingerprintManager.hasEnrolledFingerprints()) {
067                message = "No fingerprint configured.";
068                PredixSDKLogger.trace(this, message);
069                return false;
070            }
071            // Verify that the lock screen is secure, protected by PIN,password or pattern
072            if (!keyguardManager.isKeyguardSecure()) {
073                message = "Secure lock screen not enabled";
074                PredixSDKLogger.trace(this, message);
075                return false;
076            }
077
078        } catch (SecurityException se) {
079           throw new TouchIDAuthException(se);
080        }
081        message = "Fingerprint is supported and configured";
082        PredixSDKLogger.trace(this, message);
083        return true;
084
085    }
086
087    @TargetApi(Build.VERSION_CODES.M)
088    public void setupKeyAndCipher() throws TouchIDAuthException {
089        if (isFingerprintSupported()) {
090            generateKey();
091            Cipher cipher = generateCipher();
092            cryptoObject = new FingerprintManager.CryptoObject(cipher);
093        }
094
095    }
096
097
098    /**
099     * Get access to Android keystore to store the key used to encrypt/decrypt an object and then
100     * generate an encryption key
101     *
102     * @throws TouchIDAuthException
103     */
104    @TargetApi(Build.VERSION_CODES.M)
105    public void generateKey() throws TouchIDAuthException {
106        try {
107            // Get the reference to the key store
108            keyStore = KeyStore.getInstance(ANDROID_KEYSTORE_NAME);
109
110            // Key generator to generate the key
111            keyGenerator = KeyGenerator.getInstance(KeyProperties.KEY_ALGORITHM_AES, ANDROID_KEYSTORE_NAME);
112            keyStore.load(null);
113            keyGenerator.init(new KeyGenParameterSpec.Builder(KEY_NAME, KeyProperties.PURPOSE_ENCRYPT | KeyProperties.PURPOSE_DECRYPT)
114                    .setBlockModes(KeyProperties.BLOCK_MODE_CBC)
115                    .setUserAuthenticationRequired(true)
116                    .setEncryptionPaddings(KeyProperties.ENCRYPTION_PADDING_PKCS7)
117                    .build());
118            keyGenerator.generateKey();
119
120        } catch (KeyStoreException
121                | NoSuchAlgorithmException
122                | NoSuchProviderException
123                | InvalidAlgorithmParameterException
124                | CertificateException
125                | IOException exc) {
126            throw new TouchIDAuthException(exc);
127        }
128
129
130    }
131
132    /**
133     * Generate the Cipher
134     *
135     * @return
136     * @throws TouchIDAuthException
137     */
138    public Cipher generateCipher() throws TouchIDAuthException {
139        try {
140            Cipher cipher = Cipher.getInstance(KeyProperties.KEY_ALGORITHM_AES + "/" + KeyProperties.BLOCK_MODE_CBC + "/" + KeyProperties.ENCRYPTION_PADDING_PKCS7);
141            SecretKey key = (SecretKey) keyStore.getKey(KEY_NAME, null);
142            cipher.init(Cipher.ENCRYPT_MODE, key);
143            return cipher;
144        } catch (NoSuchAlgorithmException
145                | NoSuchPaddingException
146                | InvalidKeyException
147                | UnrecoverableKeyException
148                | KeyStoreException exc) {
149            throw new TouchIDAuthException(exc);
150        }
151    }
152
153    public String getMessage() {
154        return message;
155    }
156
157    public FingerprintManager getFingerprintManager() {
158        return fingerprintManager;
159    }
160
161    public FingerprintManager.CryptoObject getCryptoObject() {
162        return cryptoObject;
163    }
164}
