022public class AndroidMemChecker implements /*ComponentCallbacks2,*/ MemoryUsageChecker {
023
024    private Context context;
025    public AndroidMemChecker(Context context) {
026        this.context = context;
027    }
028
029
030    @Override
031    public Map<String, Object> getCurrentUsage() {
032        return checkMemoryPressure();
033    }
034
035    private ActivityManager.MemoryInfo getMemoryInfo() {
036        ActivityManager activityManager = (ActivityManager) this.context.getSystemService(Context.ACTIVITY_SERVICE);
037        ActivityManager.MemoryInfo memoryInfo = new ActivityManager.MemoryInfo();
038        activityManager.getMemoryInfo(memoryInfo);
039        return memoryInfo;
040    }
041
042    private Map<String, Object> checkMemoryPressure() {
043        Map<String, Object> memUsage = new HashMap<>();
044
045        ActivityManager.MemoryInfo memoryInfo = getMemoryInfo();
046        memUsage.put(RAM_AVAILABLE, memoryInfo.availMem);
047        memUsage.put(RAM_TOTAL, memoryInfo.totalMem);
048
049
050        StatFs stats = getFileSystemStats();
051        long blockSize = stats.getBlockSizeLong();
052        long availableBlocks = stats.getAvailableBlocksLong();
053        long totalBlocks = stats.getBlockCountLong();
054        long availableSpace = availableBlocks * blockSize;
055        long totalSpace = totalBlocks * blockSize;
056        memUsage.put(DISK_INTERNAL_AVAILABLE, availableSpace);
057        memUsage.put(DISK_INTERNAL_TOTAL, totalSpace);
058
059        /* TODO: Will be sent from JavaSDK
060        if (memoryInfo.lowMemory) {
061            sendNotification(MemoryNotificationLevel.Critical);
062        }*/
063
064        return memUsage;
065    }
066
067
068    private StatFs getFileSystemStats() {
069        File path = Environment.getDataDirectory();
070        StatFs space = new StatFs(path.getPath());
071        return space;
072    }
073
074    public void onTrimMemory(int trimMemoryUiHidden) {
075
076    }
077}
