001package com.ge.predix.mobile.core;
002
003import android.app.ActivityManager;
004import android.content.ComponentCallbacks2;
005import android.content.Context;
006import android.content.res.Configuration;
007
008import com.ge.predix.mobile.core.notifications.LowMemoryNotification;
009import com.ge.predix.mobile.enums.MemoryNotificationLevel;
010import com.ge.predix.mobile.logging.PredixSDKLogger;
011
012import java.util.concurrent.Executors;
013import java.util.concurrent.ScheduledExecutorService;
014import java.util.concurrent.TimeUnit;
015

020
021public class AndroidMemChecker implements ComponentCallbacks2 {
022
023    private Context context;
024    private ScheduledExecutorService executor = Executors.newScheduledThreadPool(1);
025    public AndroidMemChecker(Context context) {
026        this.context = context;
027    }
028
029    public void startMemChecker() {
030        try {
031            executor.scheduleAtFixedRate(new Runnable() {
032                @Override
033                public void run() {
034                    checkMemoryPressure();
035                }
036            }, 60, 60, TimeUnit.SECONDS);
037        } catch (Throwable t) {
038            PredixSDKLogger.error("could not start memory checker due to error: " + t);
039        }
040    }
041    
042    private void checkMemoryPressure() {
043        ActivityManager.MemoryInfo memoryInfo = getAvailableMemory();
044        if (memoryInfo.lowMemory) {
045            sendNotification(MemoryNotificationLevel.Critical);
046        }
047    }
048
049    private ActivityManager.MemoryInfo getAvailableMemory() {
050        ActivityManager activityManager = (ActivityManager) this.context.getSystemService(Context.ACTIVITY_SERVICE);
051        ActivityManager.MemoryInfo memoryInfo = new ActivityManager.MemoryInfo();
052        activityManager.getMemoryInfo(memoryInfo);
053        return memoryInfo;
054    }
055
056    @Override
057    public void onTrimMemory(int level) {
058        switch (level) {
059
060//               Release any UI objects that currently hold memory. The user interface has moved to the background.
061//               NotificationBus.defaultInstance().postNotification(new LowMemoryNotification(MemoryNotificationLevel.Normal));
062            case ComponentCallbacks2.TRIM_MEMORY_UI_HIDDEN:
063
064            case ComponentCallbacks2.TRIM_MEMORY_RUNNING_MODERATE:
065            case ComponentCallbacks2.TRIM_MEMORY_RUNNING_LOW:
066            case ComponentCallbacks2.TRIM_MEMORY_RUNNING_CRITICAL:
067//               Release any memory that your app doesn't need to run. The device is running low on memory while the app is running.
068//               The event raised indicates the severity of the memory-related event. If the event is TRIM_MEMORY_RUNNING_CRITICAL, then the system will begin killing background processes.
069                sendNotification(MemoryNotificationLevel.Warning);
070                break;
071
072            case ComponentCallbacks2.TRIM_MEMORY_BACKGROUND:
073            case ComponentCallbacks2.TRIM_MEMORY_MODERATE:
074            case ComponentCallbacks2.TRIM_MEMORY_COMPLETE:
075
076//               Release as much memory as the process can. The app is on the LRU list and the system is running low on memory.
077//               The event raised indicates where the app sits within the LRU list. If the event is TRIM_MEMORY_COMPLETE, the process will be one of the first to be terminated.
078                sendNotification(MemoryNotificationLevel.Critical);
079                break;
080
081            default:
082//                Release any non-critical data structures. The app received an unrecognized memory level value from the system. Treat this as a generic low-memory message.
083                sendNotification(MemoryNotificationLevel.Any);
084                break;
085        }
086    }
087
088    @Override
089    public void onConfigurationChanged(Configuration newConfig) {
090
091    }
092
093    @Override
094    public void onLowMemory() {
095        PredixSDKLogger.warning(this, "A critical memory warning!");
096        sendNotification(MemoryNotificationLevel.Critical);
097    }
098
099    private void sendNotification(MemoryNotificationLevel notificationLevel) {
100        PredixSDKLogger.warning(this, "Sending a low-memory notification: "+notificationLevel.toString());
101        NotificationBus.defaultInstance().postNotification(new LowMemoryNotification(notificationLevel));
102    }
103
104    public void stopMemChecker() {
105        this.executor.shutdown();
106    }
107}
