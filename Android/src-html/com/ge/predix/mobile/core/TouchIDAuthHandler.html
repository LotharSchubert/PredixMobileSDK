001package com.ge.predix.mobile.core;
002
003import android.annotation.TargetApi;
004import android.hardware.fingerprint.FingerprintManager;
005import android.os.Build;
006import android.os.CancellationSignal;
007
008
009/**
010 * Created by George Fouche on 2/16/17.
011 */
012
013@TargetApi(Build.VERSION_CODES.M)
014public class TouchIDAuthHandler extends FingerprintManager.AuthenticationCallback {
015
016    private TouchIDCallback touchIDCallback;
017    private CancellationSignal signal;
018
019    public TouchIDAuthHandler() {
020        this.signal = new CancellationSignal();
021    }
022
023    @Override
024    public void onAuthenticationError(int errorCode, CharSequence errString) {
025        super.onAuthenticationError(errorCode, errString);
026        String message = "Authentication canceled";
027        touchIDCallback.onComplete(true, false, message);
028    }
029
030    @Override
031    public void onAuthenticationHelp(int helpCode, CharSequence helpString) {
032        super.onAuthenticationHelp(helpCode, helpString);
033    }
034
035    @Override
036    public void onAuthenticationSucceeded(FingerprintManager.AuthenticationResult result) {
037        super.onAuthenticationSucceeded(result);
038        String message = "Authentication ok";
039        touchIDCallback.onComplete(true, true, message);
040
041    }
042
043    @Override
044    public void onAuthenticationFailed() {
045        super.onAuthenticationFailed();
046        String message = "Authentication failed";
047        touchIDCallback.onComplete(true, false, message);
048    }
049
050
051    @TargetApi(Build.VERSION_CODES.M)
052    public void doAuth(FingerprintManager manager, FingerprintManager.CryptoObject obj, TouchIDCallback touchIDCallback) {
053        this.touchIDCallback = touchIDCallback;
054        try {
055            manager.authenticate(obj, signal, 0, this, null);
056        } catch (SecurityException sce) {
057            touchIDCallback.onComplete(true, false, sce.getMessage());
058        }
059    }
060    @TargetApi(Build.VERSION_CODES.LOLLIPOP)
061    public void cancelAuth() {
062        if (signal !=null) {
063            signal.cancel();
064        }
065    }
066}
