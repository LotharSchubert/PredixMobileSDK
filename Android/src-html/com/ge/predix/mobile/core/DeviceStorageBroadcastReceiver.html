001package com.ge.predix.mobile.core;
002
003import android.content.BroadcastReceiver;
004import android.content.Context;
005import android.content.Intent;
006
007import com.ge.predix.mobile.core.notifications.DiskSpaceNotification;
008import com.ge.predix.mobile.enums.DiskSpaceNotificationType;
009import com.ge.predix.mobile.logging.PredixSDKLogger;
010
011/**
012 * Created by 212460388 on 6/21/17.
013 * Copyright Â© 2017 GE. All rights reserved.
014 * This class acts as a {@link BroadcastReceiver} for device storage space level notifications.
015 */
016public class DeviceStorageBroadcastReceiver extends BroadcastReceiver {
017    @Override
018    public void onReceive(Context context, Intent intent) {
019        String action = intent.getAction();
020        if(action.equals(Intent.ACTION_DEVICE_STORAGE_LOW)){
021//            TODO: Can we get some info from Intent?? Need to check on it.
022            sendNotification(new DiskSpaceNotification());
023        } else if(action.equals(Intent.ACTION_DEVICE_STORAGE_OK)) {
024            sendNotification(new DiskSpaceNotification(DiskSpaceNotificationType.OK));
025        } else {
026
027        }
028    }
029
030    private void sendNotification(DiskSpaceNotification notification) {
031        PredixSDKLogger.warning(this, "Sending a low storage notification.");
032        NotificationBus.defaultInstance().postNotification(notification);
033    }
034}
