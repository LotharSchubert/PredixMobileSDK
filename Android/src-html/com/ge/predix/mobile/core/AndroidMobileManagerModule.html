001package com.ge.predix.mobile.core;
002
003import android.content.Context;
004
005import com.ge.predix.mobile.platform.CookieStorage;
006import com.ge.predix.mobile.platform.PlatformContext;
007import com.ge.predix.mobile.platform.WindowView;
008import com.ge.predix.mobile.serviceFoundation.ServiceRouter;
009import com.ge.predix.mobile.services.OrientationChangeManager;
010import com.ge.predix.mobile.services.OrientationChangeManagerImpl;
011import com.ge.predix.mobile.services.ProximityManager;
012import com.ge.predix.mobile.services.ProximityManagerImpl;
013import com.ge.predix.mobile.util.ConfigurationFileManager;
014import com.ge.predix.mobile.util.SecureStorageRequirements;
015
016import java.io.File;
017
018import javax.inject.Singleton;
019
020import dagger.Module;
021import dagger.ObjectGraph;
022import dagger.Provides;
023
024/**
025 * Created by Clifton Craig on 8/25/16.
026 * Copyright GE 8/25/16
027 */
028@Module(
029        library = true,
030        includes = MobileManagerModule.class,
031        injects = MobileManager.class,
032        overrides = true
033)
034
035public class AndroidMobileManagerModule {
036    private final Context androidContext;
037    private final ConfigurationManager configurationPreferences;
038    private final ReachabilityManager reachabilityManager;
039    private final ConnectionTypeProvider connectionTypeProvider;
040    private final PredixConnectionManager predixConnectionManager;
041    private final PredixLocalNotificationManager localNotificationManager;
042    private final SecureStorageRequirements secureStorageReq;
043    private ViewInterface viewInterface;
044    private ObjectGraph graph;
045
046    public AndroidMobileManagerModule(Context context, ViewInterface viewInterface) {
047        this.viewInterface = viewInterface;
048        this.androidContext = context;
049        this.reachabilityManager = new ReachabilityManager();
050        this.connectionTypeProvider = new AndroidConnectivityTypeProvider(this.androidContext);
051        this.predixConnectionManager = new PredixConnectionManager();
052        this.predixConnectionManager.setConnectionTypeProvider(this.connectionTypeProvider);
053        this.configurationPreferences = new ConfigurationManager(viewInterface.getDefaultPreferences(), androidContext);
054        this.localNotificationManager = new PredixLocalNotificationManagerImpl(this.androidContext);
055        this.secureStorageReq = new SecureStorageRequirementsImpl(this.androidContext);
056    }
057
058    @Provides
059    public PredixConnectionManager providesConnectionManager() {
060        return predixConnectionManager;
061    }
062
063    @Provides
064    public ConfigurationFileManager providesConfigCacheFile(Context context) {
065        return new ConfigurationFileManager(new File(context.getCacheDir(), "cachefile.properties"));
066    }
067
068    @Provides
069    public PlatformContext providesCBContext() {
070        return viewInterface.getContext();
071    }
072
073    @Provides
074    public AuthHandler providesAuthHandler() {
075        return viewInterface.getAuthHandler();
076    }
077
078    @Provides
079    public WindowView providesWindowView() {
080        return viewInterface.getWindowView();
081    }
082
083    @Provides
084    public Context providesAndroidContext() {
085        return androidContext;
086    }
087
088    @Provides
089    public ConnectionTypeProvider providesConnectionTypeProvider() {
090        return connectionTypeProvider;
091    }
092
093    @Provides
094    @Singleton
095    public ProximityManager providesProximitySensorService() {
096        return new ProximityManagerImpl(androidContext);
097    }
098
099    @Provides
100    public OrientationChangeManager providesOrientationChangeManager() {
101        return OrientationChangeManagerImpl.getInstance();
102    }
103
104    @Provides
105    @Singleton
106    public CookieStorage providesCookieStorage() {
107        HTTPCookieStorage cookieStorage = new AndroidCookieStorage();
108        cookieStorage.clearSessionCookies();
109        return cookieStorage;
110    }
111
112    @Provides
113    public ReachabilityManager providesReachabilityManager() {
114        return reachabilityManager;
115    }
116
117    @Provides
118    @Singleton
119    public ServiceRouter providesServiceRouter() {
120        ServicesModule servicesModule = new ServicesModule();
121        ObjectGraph newGraph = this.graph.plus(new Object[]{servicesModule});
122        servicesModule.setObjectGraph(newGraph);
123        DaggerServiceProvider daggerServiceProvider = new DaggerServiceProvider();
124        newGraph.inject(daggerServiceProvider);
125        ServiceRouterImpl serviceRouter = new ServiceRouterImpl();
126        serviceRouter.addServiceProvider(daggerServiceProvider);
127        return serviceRouter;
128    }
129
130    @Provides
131    public ConfigurationPreferences providesConfigurationPreference() {
132        return this.configurationPreferences;
133    }
134
135    @Provides
136    public PredixLocalNotificationManager providesLocalNotificationManager() {
137        return this.localNotificationManager;
138    }
139
140    @Provides
141    public TouchID providesTouchID(AndroidTouchID androidTouchID) {
142        return androidTouchID;
143    }
144
145    @Provides
146    @Singleton
147    public SecureStorageRequirements providesSecureStorageRequirements() {
148        return this.secureStorageReq;
149    }
150
151    public void setGraph(ObjectGraph graph) {
152        this.graph = graph;
153    }
154}
