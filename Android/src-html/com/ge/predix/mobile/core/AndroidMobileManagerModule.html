001package com.ge.predix.mobile.core;
002
003import android.content.Context;
004
005import com.ge.predix.mobile.platform.CookieStorage;
006import com.ge.predix.mobile.platform.MapConfigurationPreferences;
007import com.ge.predix.mobile.platform.PMAssetsManager;
008import com.ge.predix.mobile.platform.PlatformContext;
009import com.ge.predix.mobile.platform.WindowView;
010import com.ge.predix.mobile.serviceFoundation.ServiceRouter;
011import com.ge.predix.mobile.services.OrientationChangeManager;
012import com.ge.predix.mobile.services.OrientationChangeManagerImpl;
013import com.ge.predix.mobile.services.ProximityManager;
014import com.ge.predix.mobile.services.ProximityManagerImpl;
015import com.ge.predix.mobile.util.ConfigurationFileManager;
016import com.ge.predix.mobile.util.SecureStorageRequirements;
017
018import java.io.File;
019import java.lang.ref.WeakReference;
020import java.util.Map;
021
022import javax.inject.Singleton;
023
024import dagger.Module;
025import dagger.ObjectGraph;
026import dagger.Provides;
027
028/**
029 * Created by Clifton Craig on 8/25/16.

031 */
032@Module(
033        library = true,
034        includes = MobileManagerModule.class,
035        injects = MobileManager.class,
036        overrides = true
037)
038
039public class AndroidMobileManagerModule {
040    private final Context androidContext;
041    private WeakReference<ViewInterface> viewInterface;
042    private ObjectGraph graph;
043    private PredixConnectionManager predixConnectionManager;
044    private WeakReference<AuthenticationManager> authenticationManager;
045
046    public AndroidMobileManagerModule(Context context) {
047        this.androidContext = context;
048
049        PlatformImpl.getInstance().setCookieStorage(new AndroidCookieStorage());
050        PlatformImpl.getInstance().setLocalNotificationManager(this.providesLocalNotificationManager());
051        PlatformImpl.getInstance().setDbManager(PredixCouchbaseManager.defaultInstance(CoreConfigurationLoader.getWipeSetting()));
052    }
053
054    @Provides
055    public ConfigurationFileManager providesConfigCacheFile(Context context) {
056        return new ConfigurationFileManager(new File(context.getCacheDir(), "cachefile.properties"));
057    }
058
059    @Provides
060    public PlatformContext providesCBContext() {
061        ViewInterface viewInterface = this.viewInterface.get();
062        if (viewInterface == null) {
063            return null;
064        }
065        return viewInterface.getContext();
066    }
067
068    @Provides
069    public WindowView providesWindowView() {
070        ViewInterface viewInterface = this.viewInterface.get();
071        if (viewInterface == null) {
072            return null;
073        }
074
075        return viewInterface.getWindowView();
076    }
077
078    @Provides
079    public Context providesAndroidContext() {
080        return androidContext;
081    }
082
083    @Provides
084    @Singleton
085    public ConnectionTypeProvider providesConnectionTypeProvider() {
086        return new AndroidConnectivityTypeProvider(this.androidContext);
087    }
088
089    @Provides
090    @Singleton
091    public ProximityManager providesProximitySensorService() {
092        return new ProximityManagerImpl(androidContext);
093    }
094
095    @Provides
096    public OrientationChangeManager providesOrientationChangeManager() {
097        return OrientationChangeManagerImpl.getInstance();
098    }
099
100
101    @Provides
102    @Singleton
103    public CookieStorage providesCookieStorage() {
104        HTTPCookieStorage cookieStorage = new AndroidCookieStorage();
105        cookieStorage.clearSessionCookies();
106        return cookieStorage;
107    }
108
109    @Provides
110    @Singleton
111    public ReachabilityManager providesReachabilityManager() {
112        return new ReachabilityManager();
113    }
114
115    @Provides
116    @Singleton
117    public ServiceRouter providesServiceRouter() {
118        ServicesModule servicesModule = new ServicesModule();
119        ObjectGraph newGraph = this.graph.plus(new Object[]{servicesModule});
120        servicesModule.setObjectGraph(newGraph);
121        DaggerServiceProvider daggerServiceProvider = new DaggerServiceProvider();
122        newGraph.inject(daggerServiceProvider);
123        ServiceRouterImpl serviceRouter = new ServiceRouterImpl();
124        serviceRouter.addServiceProvider(daggerServiceProvider);
125        return serviceRouter;
126    }
127
128    @Provides
129    @Singleton
130    public ConfigurationPreferences providesConfigurationPreference(Platform platform) {
131        return new ConfigurationManager(getPreferences(), androidContext, platform);
132    }
133
134    @Provides
135    @Singleton
136    public UserSettingsPreference providesUserSettingsPreference(NotificationBus notificationSender) {
137        return new AndroidUserSettingsPreference(getPreferences(), androidContext, notificationSender);
138    }
139
140    @Provides
141    @Singleton
142    public PredixLocalNotificationManager providesLocalNotificationManager() {
143        return new PredixLocalNotificationManagerImpl(this.androidContext);
144    }
145
146    @Provides
147    @Singleton
148    public PMAssetsManager providesPMAssetsManager() {
149        return new PMAssetsManagerImpl(this.androidContext);
150    }
151
152    @Provides
153    public TouchID providesTouchID(AndroidTouchID androidTouchID) {
154        return androidTouchID;
155    }
156
157    @Provides
158    @Singleton
159    public SecureStorageRequirements providesSecureStorageRequirements() {
160        return new SecureStorageRequirementsImpl(this.androidContext);
161    }
162
163    @Provides
164    public MemoryUsageChecker providesMemoryUsageChecker() {
165        return new AndroidMemChecker(this.androidContext);
166    }
167
168    @Provides
169    public AuthenticationManager providesAuthenticationManager() {
170        if (authenticationManager == null) {
171            return null;
172        }
173
174        return authenticationManager.get();
175    }
176
177    @Provides
178    @Singleton
179    public PredixConnectionManager providesPredixConnectionManager(AuthenticationManager authenticationManager, ConnectionTypeProvider connectionTypeProvider) {
180        if (null == predixConnectionManager) {
181            predixConnectionManager = new PredixConnectionManager();
182            predixConnectionManager.authenticationManager = authenticationManager;
183        }
184
185        predixConnectionManager.setConnectionTypeProvider(connectionTypeProvider);
186        return predixConnectionManager;
187    }
188
189    public void setGraph(ObjectGraph graph) {
190        this.graph = graph;
191    }
192
193    public void setViewInterface(ViewInterface viewInterface) {
194        this.viewInterface = new WeakReference<>(viewInterface);
195
196        if (PlatformImpl.getInstance().context() == null) {
197            PlatformImpl.getInstance().setContext(viewInterface.getContext());
198        }
199        if (PlatformImpl.getInstance().windowView() == null) {
200            PlatformImpl.getInstance().setWindowView(viewInterface.getWindowView());
201        }
202
203        PlatformImpl.getInstance().setCookieStorage(new AndroidCookieStorage());
204        PlatformImpl.getInstance().setLocalNotificationManager(this.providesLocalNotificationManager());
205
206        //TODO: refactor this so that when the view interface is given we can make sure ConfigurationPreferencesProvider has already been set so we don't need the check
207        if (PredixMobileConfiguration.ConfigurationPreferencesProvider == null) {
208            PredixMobileConfiguration.ConfigurationPreferencesProvider = new MapConfigurationPreferences(viewInterface.getDefaultPreferences());
209        } else {
210            PredixMobileConfiguration.ConfigurationPreferencesProvider.putAll(viewInterface.getDefaultPreferences());
211        }
212    }
213
214    public void setAuthenticationManager(AuthenticationManager authenticationManager) {
215        this.authenticationManager = new WeakReference<>(authenticationManager);
216    }
217
218    private Map<String, Object> getPreferences() {
219        ViewInterface viewInterface = this.viewInterface.get();
220        if (viewInterface == null) {
221            return null;
222        }
223
224        return viewInterface.getDefaultPreferences();
225    }
226}
