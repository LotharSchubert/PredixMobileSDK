001package com.ge.predix.mobile.core;
002
003import android.app.Notification;
004import android.app.PendingIntent;
005import android.content.Context;
006import android.content.Intent;
007import android.os.Bundle;
008import android.support.v7.app.NotificationCompat;
009
010import com.ge.predix.mobile.logging.PredixSDKLogger;
011
012import javax.inject.Inject;
013
014/**
015 * Created by 212460388 on 9/15/16.
016 */
017public final class NotificationCreatorImpl implements NotificationCreator {
018
019    @Inject
020    Class/*<? extends Activity>*/ activityClass;
021
022    private final int requestCode = 91916; //09-19-2016
023
024    @Override
025    public final Notification create(final Context context, final Intent intent, final String NOTIFICATION_PROMPT_KEY, final String NOTIFICATION_ADDITIONAL_KEY) {
026
027        if (null == intent) {
028            PredixSDKLogger.error(this, "intent param is missing!");
029            return null;
030        }
031        if (null == context) {
032            PredixSDKLogger.error(this, "context param is missing!");
033            return null;
034        }
035        if (null == this.activityClass) {
036            this.activityClass = context.getClass();
037        }
038
039        Bundle bundle = intent.getExtras();
040        intent.setClass(context, activityClass);
041
042        intent.setFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP
043                | Intent.FLAG_ACTIVITY_SINGLE_TOP);
044
045        PendingIntent pendingIntent = PendingIntent.getActivity(context, requestCode, intent, PendingIntent.FLAG_UPDATE_CURRENT | PendingIntent.FLAG_ONE_SHOT);
046
047        NotificationCompat.Builder builder = new NotificationCompat.Builder(context);
048        String title = null;
049        String contentText = null;
050        if (null != bundle) {
051
052            title = bundle.getString(NOTIFICATION_PROMPT_KEY);
053            contentText = bundle.getString(NOTIFICATION_ADDITIONAL_KEY);
054        }
055        if (null == title) {
056            title = "Predix Mobile Reference app";
057        }
058        if (null == contentText) {
059            contentText = "Predix Mobile Reference app";
060        }
061
062        Notification notification = builder.setContentTitle(title)
063                .setContentText(contentText)
064                .setAutoCancel(true) // dismiss the notification from the notification area on tap
065                .setTicker(title)
066                .setSmallIcon(android.R.drawable.ic_menu_camera)
067                .setContentIntent(pendingIntent).build();
068
069        notification.flags |= Notification.FLAG_ONGOING_EVENT;
070        notification.flags |= Notification.FLAG_SHOW_LIGHTS;
071        notification.flags |= Notification.FLAG_AUTO_CANCEL;
072        return notification;
073    }
074}
