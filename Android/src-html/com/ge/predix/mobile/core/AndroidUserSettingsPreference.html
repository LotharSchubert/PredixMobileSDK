001package com.ge.predix.mobile.core;
002
003import android.content.Context;
004import android.content.SharedPreferences;
005import android.preference.PreferenceManager;
006
007import com.ge.predix.mobile.core.notifications.PreferencesChangedNotification;
008import com.ge.predix.mobile.logging.PredixSDKLogger;
009import com.ge.predix.mobile.platform.NotificationSender;
010
011import org.json.JSONArray;
012import org.json.JSONObject;
013
014import java.util.Map;
015

020public class AndroidUserSettingsPreference implements UserSettingsPreference, SharedPreferences.OnSharedPreferenceChangeListener {
021    private final SharedPreferences sharedPreferences;
022    private Platform platform;
023    private NotificationSender notificationSender;
024
025    public AndroidUserSettingsPreference(Map<String, Object> defaultPreferences, Context context, NotificationSender notificationSender) {
026        this.notificationSender = notificationSender;
027        if (context != null) {
028            this.sharedPreferences = PreferenceManager.getDefaultSharedPreferences(context);
029            if (this.sharedPreferences != null) {
030                copyDefaultsToShared(defaultPreferences, this.sharedPreferences);
031                this.sharedPreferences.registerOnSharedPreferenceChangeListener(this);
032            } else {
033                PredixSDKLogger.error(this, "getDefaultSharedPreferences is null, shared preferences was not instantiated correctly.");
034            }
035        }
036        else {
037            PredixSDKLogger.error(this, "Context is null, shared preferences was not instantiated correctly.");
038            sharedPreferences = null;
039        }
040    }
041
042    @Override
043    public Object getPreference(String key) {
044        if (null == this.sharedPreferences)
045        {
046            PredixSDKLogger.error(this, "Shared Preferences was not instantiated correctly.");
047            return null;
048        }
049        if(this.sharedPreferences.contains(key)) {
050            return this.sharedPreferences.getAll().get(key);
051        }
052        return null;
053    }
054
055    @Override
056    public void setPreference(String key, Object value) {
057        storePreference(key, value);
058    }
059
060    @Override
061    public void deletePreference(String key) {
062        PredixSDKLogger.trace(this, String.format("Deleting key %s from shared preferences.", key));
063
064        SharedPreferences.Editor editor = this.sharedPreferences.edit();
065        editor.remove(key);
066        editor.apply();
067    }
068
069    @Override
070    public Map<String, Object> getAllPreferences() {
071        PredixSDKLogger.trace(this, "Fetching all key/values from shared preferences.");
072
073        @SuppressWarnings("unchecked")
074        Map<String, Object> preferences = (Map<String, Object>) this.sharedPreferences.getAll();
075
076        return preferences;
077    }
078
079    @Override
080    public void deleteAllPreferences() {
081        PredixSDKLogger.trace(this, "Deleting all keys from shared preferences.");
082        SharedPreferences.Editor editor = this.sharedPreferences.edit();
083        editor.clear();
084        editor.apply();
085    }
086
087    private void storePreference(String key, Object val) {
088
089        PredixSDKLogger.trace(this, String.format("Adding/updating key %s to shared preferences.", key));
090        SharedPreferences.Editor editor = this.sharedPreferences.edit();
091
092        if (val instanceof Integer) {
093            editor.putInt(key, (Integer) val);
094        }
095        else if (val instanceof Long ) {
096            editor.putLong(key, (Long) val);
097        }
098        else if (val instanceof Double ) {
099            editor.putFloat(key, ((Double) val).floatValue()); //no double method in Editor, so using Float.
100        }
101        else if (val instanceof Float) {
102            editor.putFloat(key, (Float) val);
103        }
104        else if (val instanceof Boolean) {
105            editor.putBoolean(key, (Boolean) val);
106        }
107        else if (val instanceof JSONObject) {
108            editor.putString(key, val.toString());
109        }
110        else if (val instanceof JSONArray) {
111            editor.putString(key, val.toString());
112        }
113        else if (JSONObject.NULL.equals(val)) {
114            editor.putString(key, "NULL");
115        }
116        else {
117            String value = (String) val;
118            editor.putString(key, value);
119        }
120
121        editor.apply();
122    }
123
124    private void copyDefaultsToShared(Map<String, Object> defaultPreferences, SharedPreferences sharedPreferences) {
125
126        SharedPreferences.Editor editor = sharedPreferences.edit();
127        Boolean hadChanges = false;
128
129        for (Map.Entry<String, Object> preference : defaultPreferences.entrySet()) {
130
131            String key = preference.getKey();
132            if(!sharedPreferences.contains(preference.getKey())) {
133                hadChanges = true;
134                Object value = preference.getValue();
135
136                if (value instanceof String) {
137                    editor.putString(key, (String)value);
138                } else if (value instanceof Integer) {
139                    editor.putInt(key, ((Integer) value));
140                } else if (value instanceof Float) {
141                    editor.putFloat(key, ((Float) value));
142                } else if (value instanceof Long) {
143                    editor.putLong(key, ((Long) value));
144                } else if (value instanceof Boolean) {
145                    editor.putBoolean(key, ((Boolean) value));
146                } else {
147                    PredixSDKLogger.debug(this, String.format("Unhandled data type \"%s\" storing value as String", value.getClass().getCanonicalName()));
148                    editor.putString(key, String.valueOf(value));
149                }
150            }
151        }
152
153        if(hadChanges){
154            editor.apply();
155        }
156    }
157
158    @Override
159    public void onSharedPreferenceChanged(SharedPreferences sharedPreferences, String key) {
160        // post notification to notifcation bus anytime preference is changed
161        PreferencesChangedNotification preferencesChangedNotification = new PreferencesChangedNotification();
162        notificationSender.postNotification(preferencesChangedNotification);
163    }
164}
