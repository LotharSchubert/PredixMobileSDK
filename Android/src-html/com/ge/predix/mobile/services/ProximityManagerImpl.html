002
003import android.content.Context;
004import android.hardware.Sensor;
005import android.hardware.SensorEvent;
006import android.hardware.SensorEventListener;
007import android.hardware.SensorManager;
008
009import com.ge.predix.mobile.core.NotificationBus;
010import com.ge.predix.mobile.logging.PredixSDKLogger;
011
012import org.json.JSONException;
013import org.json.JSONObject;
014
015import java.util.Arrays;
016
017/**
018 * Created by 212460388 on 3/1/17.
019 */
020public class ProximityManagerImpl implements SensorEventListener, ProximityManager{
021
022    private final Context context;
023    private SensorManager mSensorManager;
024    private Sensor mProximitySensor;
025    private SensorEvent mLastKnownEvent;
026
027    enum SensorState {
028        STOPPED, INITIALIZING, STARTED
029    }
030    private SensorState state = SensorState.STOPPED;
031    private ProximityCallback mPendingCallback;
032
033    public ProximityManagerImpl(Context context) {
034        this.context = context;
035    }
036
037    /**
038     *
039     * @return
040     * @category ProximityManager
041     */
042    @Override
043    public void getProximity(ProximityCallback callback) {
044        if (state == SensorState.STARTED) {
045            callback.onComplete(true, sensorEventToJSON(this.mLastKnownEvent), null);
046        }
047        else if (state == SensorState.INITIALIZING) {
048            callback.onComplete(false, null, "Processing older request");
049        }
050        else {
051            state = SensorState.INITIALIZING;
052            mPendingCallback = callback;
053            startSensing();
054        }
055
056    }
057
058    /**
059     *
060     * @param callback
061     * @category ProximityManager
062     */
063    @Override
064    public void stopProximity(ProximityCallback callback) {
065        if (state == SensorState.STARTED) {
066            stopSensing();
067            this.mLastKnownEvent = null;
068            state = SensorState.STOPPED;
069            callback.onComplete(true, null, null);
070        }
071        else {
072//            TODO: Do we want to notify that its already stopped.
073            callback.onComplete(true, null, null);
074        }
075
076    }
077
078    private void startSensing() {
079        if (null == this.mSensorManager) {
080            mSensorManager = (SensorManager) this.context.getSystemService(Context.SENSOR_SERVICE);
081            mProximitySensor = mSensorManager.getDefaultSensor(Sensor.TYPE_PROXIMITY);
082        }
083        if (null == mProximitySensor) {
084            this.mPendingCallback.onComplete(false, null, "Sensor missing!");
085            this.mPendingCallback = null;
086        }
087        else {
088            mSensorManager.registerListener(this, mProximitySensor, SensorManager.SENSOR_DELAY_NORMAL);
089            state = SensorState.STARTED;
090        }
091
092    }
093
094    private void stopSensing() {
095        mSensorManager.unregisterListener(this);
096    }
097
098    private JSONObject sensorEventToJSON(SensorEvent sensorEvent) {
099        JSONObject toReturn = new JSONObject();
100        try {
101            toReturn.put("values", Arrays.toString(sensorEvent.values));
102            toReturn.put("accuracy",sensorEvent.accuracy);
103            toReturn.put("timestamp",sensorEvent.timestamp);
104//            toReturn.put("all",sensorEvent.toString());
105            float distance = sensorEvent.values[0];
106            String isClose = "isClose";
107            if (distance == 0) {
108                toReturn.put(isClose,true);
109            }
110            else {
111                toReturn.put(isClose,false);
112            }
113//            for (int i = 0; i < sensorEvent.values.length; i++) {
114//                toReturn.put("value_"+i,sensorEvent.values[i]);
115//            }
116        } catch (JSONException e) {
117            PredixSDKLogger.error(this,"Error while converting sensor data: "+e.getLocalizedMessage(),e);
118        }
119
120        return toReturn;
121    }
122
123
124    /**
125     * @param event
126     * @category SensorEventListener
127     */
128    @Override
129    public void onSensorChanged(SensorEvent event) {
130        this.mLastKnownEvent = event;
131        if (state != SensorState.STARTED) {
132            return;
133        }
134        JSONObject payload = sensorEventToJSON(event);
135        if (null != this.mPendingCallback) {
136            mPendingCallback.onComplete(true, payload, null);
137            mPendingCallback = null;
138        }
139        NotificationBus.defaultInstance().postNotification(new ProximityNotification(sensorEventToJSON(event)));
140    }
141
142    /**
143     *
144     * @param sensor
145     * @param accuracy
146     * @category SensorEventListener
147     */
148    @Override
149    public void onAccuracyChanged(Sensor sensor, int accuracy) {
150
151    }
152
153}
